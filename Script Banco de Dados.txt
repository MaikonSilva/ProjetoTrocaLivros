CREATE DATABASE trocalivro;

USE trocalivro;

CREATE TABLE USUARIO(
N_COD_USUARIO INT PRIMARY KEY AUTO_INCREMENT,
V_NOME VARCHAR(70) NOT NULL,
V_LOGIN VARCHAR(18) UNIQUE NOT NULL,
V_SENHA VARCHAR(18) NOT NULL,
V_EMAIL VARCHAR(50) NOT NULL,
V_CPF VARCHAR(15) UNIQUE NOT NULL,
V_TELEFONE VARCHAR(15),
V_CELULAR VARCHAR(15),
V_CEP VARCHAR(15) NOT NULL,
V_UF VARCHAR(2),
V_CIDADE VARCHAR(50),
V_BAIRRO VARCHAR(50),
V_FOTO VARCHAR(300),
V_IDADE INT,
N_TIPO_USUARIO INT,
D_DATA_CADASTRO DATETIME,
D_DATA_ULTIMO_LOGIN DATETIME,
B_ATIVO VARCHAR(1)
);
